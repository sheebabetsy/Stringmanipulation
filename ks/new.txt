void frequency()	
{

printf("\n Enter a character\t");

void arrange()
{

for(c=0;c<n;c++)
{
	ch1=str[c]-'a';
	no[ch1]++;
}
t=0;
for(ch1='a';ch1<='z';ch1++)
{
 x=ch1-'a';
  for(c=0;c<no[x];c++)
   {
